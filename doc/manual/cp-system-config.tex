\section{System Cofiguration File Syntax}
\begin{syntdiag}
\tok{OPERATOR} \tok{CONSOLE} <rdev>
\end{syntdiag}

\begin{syntdiag}
\tok{OPERATOR} \tok{USERID} <userid>
\end{syntdiag}

\begin{syntdiag}
\tok{RDEV} <rdev> <type>
\end{syntdiag}

\begin{syntdiag}
\tok{LOGO} \tok{LOCAL} <type> <filename> <filetype>
\end{syntdiag}
